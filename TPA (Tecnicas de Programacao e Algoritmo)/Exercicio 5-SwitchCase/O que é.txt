SwitchCase serve para deixar o if e else mais simples e menos longo

package com.mycompany.switchcase;
//Import scanner
import java.util.Scanner;
/**
 Author:Derik Soares Batinga
 */ 
public class SwitchCase {
      public static void main(String[] args) {
     //Criação do scanner
         Scanner scanner=new Scanner(System.in);
        //Variáveis
         System.out.println("Digite a sua idade: ");
         int idade=scanner.nextInt();
        //Código a executar
         switch(idade){
             case 9,10:
             System.out.print("Você uma idade igual a 9 ou 10: ");
         break;
             case 11,12:
             System.out.print("Você uma idade de 11 ou 12: ");
         break;
              case 13,14:
              System.out.print("Você uma idade de 11 ou 12: ");
         break;
              case 15,16:
              System.out.print("Você uma idade de 11 ou 12: ");
         break;
              case 17,18:
              System.out.print("Você uma idade de 11 ou 12: ");
                break;
              default: System.out.print("Você possui maior idade");
       }
}
